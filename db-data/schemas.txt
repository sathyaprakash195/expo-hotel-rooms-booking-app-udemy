
// Hotels table creation

create table public.hotels (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  name text null,
  description text null,
  city text null,
  address text null,
  email text null,
  phone text null,
  images text[] null,
  status text null,
  owner_id bigint null,
  amenities text[] null,
  starting_rent double precision null,
  constraint hotels_pkey primary key (id),
  constraint hotels_owner_id_fkey foreign KEY (owner_id) references user_profiles (id)
) TABLESPACE pg_default;



// Rooms table creation

create table public.rooms (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  hotel_id bigint null,
  owner_id bigint null,
  name text null,
  description text null,
  type text null,
  rent_per_day double precision null,
  status text null,
  amenities text[] null,
  images text[] null,
  constraint rooms_pkey primary key (id),
  constraint rooms_hotel_id_fkey foreign KEY (hotel_id) references hotels (id),
  constraint rooms_owner_id_fkey foreign KEY (owner_id) references user_profiles (id)
) TABLESPACE pg_default;
